## Install

```bash
npm install --save material-ui-table
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'material-ui-table'
import 'material-ui-table/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
